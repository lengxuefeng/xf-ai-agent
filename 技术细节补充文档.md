#### SSE 事件流处理

```python
def to_sse(data: dict) -> str:
    """将字典格式化为 Server-Sent Event (SSE) 字符串"""
    return f"data: {json.dumps(data, ensure_ascii=False)}\n\n"

def stream_text(text: str, delay: float = 0.03) -> Generator[str, None, None]:
    """实现打字机效果的逐字符流式输出"""
    for char in text:
        # 中文字符和标点符号加速处理
        current_delay = delay * 0.5 if ord(char) > 127 else delay
        
        yield to_sse({
            "type": "stream", 
            "content": char
        })
        
        if current_delay > 0:
            time.sleep(current_delay)
```

### 4. 工具集成架构

#### 工具节点构建

```python
def create_tool_agent_executor(model: BaseChatModel, tools: List, state_class: Type[dict]):
    def agent_node(state):
        # 检查是否需要工具调用
        if isinstance(state["messages"][-1], ToolMessage):
            response = model.invoke(state["messages"])
        else:
            model_with_tools = model.bind_tools(tools)
            response = model_with_tools.invoke(state["messages"])
        return {"messages": [response]}
    
    def router(state):
        last_message = state["messages"][-1]
        if hasattr(last_message, "tool_calls") and last_message.tool_calls:
            return "tools"
        return END
    
    # 构建图结构
    graph = StateGraph(state_class)
    graph.add_node("agent", agent_node)
    graph.add_node("tools", ToolNode(tools))
    graph.set_entry_point("agent")
    graph.add_conditional_edges("agent", router)
    graph.add_edge("tools", "agent")
    
    return graph.compile()
```

### 6. 性能优化策略

#### 图实例复用

```python
class GraphRunner:
    def __init__(self, model_config: dict = None):
        self.model_config = model_config or {}
        self.graph = None  # 延迟初始化
        self._config_hash = None  # 配置哈希值
    
    def stream_run(self, user_input: str, session_id: str, model_config: dict = None):
        final_config = {**self.model_config, **(model_config or {})}
        config_hash = hash(json.dumps(final_config, sort_keys=True))
        
        # 只有配置变更时才重建图
        if self.graph is None or config_hash != self._config_hash:
            self.graph = create_graph(final_config)
            self._config_hash = config_hash
```

#### 并发处理优化

```python
class ConcurrentExecutor:
    def __init__(self, max_threads: int = 4):
        self.executor = ThreadPoolExecutor(max_workers=max_threads)
    
    def execute_parallel(self, tasks: List[Callable]) -> List[Any]:
        """并发执行多个任务"""
        futures = [self.executor.submit(task) for task in tasks]
        return [future.result() for future in futures]
```

```python
class RedisManager:
    def save_graph_state(self, state: dict, session_id: str, subgraph_id: str):
        """保存图状态到 Redis"""
        key = f"{subgraph_id}:{session_id}"
        serialized_state = self._serialize_state(state)
        self.redis_client.setex(key, 3600, serialized_state)  # 1小时过期
    
    def load_graph_state(self, session_id: str, subgraph_id: str) -> dict:
        """从 Redis 加载图状态"""
        key = f"{subgraph_id}:{session_id}"
        serialized_state = self.redis_client.get(key)
        return self._deserialize_state(serialized_state) if serialized_state else None
```

#### API 限流

```python
from fastapi import HTTPException
from time import time

class RateLimiter:
    def __init__(self, max_requests: int = 60, window_seconds: int = 60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = {}
    
    def check_rate_limit(self, client_ip: str):
        """检查请求频率限制"""
        now = time()
        if client_ip not in self.requests:
            self.requests[client_ip] = []
        
        # 清理过期请求
        self.requests[client_ip] = [
            req_time for req_time in self.requests[client_ip]
            if now - req_time < self.window_seconds
        ]
        
        if len(self.requests[client_ip]) >= self.max_requests:
            raise HTTPException(status_code=429, detail="请求过于频繁")
        
        self.requests[client_ip].append(now)
```

**症状**: 提示 "模型加载失败，使用默认模型"

**原因**: API 密钥配置错误或网络问题

**解决**:

```bash
     https://openrouter.ai/api/v1/models
```

#### 2. 流式响应中断

**症状**: 前端显示部分响应后停止

**原因**: 网络超时或服务器错误

**解决**:
```python
# 增加超时设置
```
